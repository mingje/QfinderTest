def qfinder_ip_sort():
    fun_name = sys._getframe().f_code.co_name
    print("*** Start to " + fun_name + " ***")
    # open qfinder
    open_qfinder()
    click()
    print("click IP address field")
    wait(2)

    s = Region(Region(402,280,109,375))
    ip_str = s.text()
    ip_list = ip_str.splitlines()
    print("Initial list: " + str(ip_list))
    # rm space
    iplist = []
    for i in ip_list:
        i = i.split("(")
        i = i[0]
        q = i.replace(' ','')
        q = q.replace('l','1')
        q = q.replace('S','5')
        iplist.append(q)
    print("Switch list: " + str(iplist))
    print("Sorting...")
    lot = map(inet_aton, iplist)
    print("Sort lot")
    lot1 = map(inet_aton, iplist)
    print("Sort lot1")
    lot.sort()
    lot1.sort(reverse=True)
    iplist1 = map(inet_ntoa, lot)
    iplist2 = map(inet_ntoa, lot1)
    print("Sorted list: " + str(iplist1))
    print("Sorted list: " + str(iplist2))
    if iplist == []:
        print("list fail")
        flag = "False"
    elif iplist == iplist1 or iplist == iplist2:
        print("pass")
        flag = "True"
    else:
        print("FAIL")
        flag = "False"
    with open("result.txt", "w") as fp:
       fp.write(flag)
    print("--- End " + fun_name + " ---")

if __name__ == "__main__":
    qfinder_ip_sort()
