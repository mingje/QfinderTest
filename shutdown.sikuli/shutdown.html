def qfinder_shutdown():
    fun_name = sys._getframe().f_code.co_name
    print("*** Start to " + fun_name + " ***")
    # open qfinder
    open_qfinder()
    #  find target NAS
    find_target_nas(name = target["name"], lanip1 = target["lanip1"])
    wait(2)
    doubleClick()
    print("click tools option")
    wait(1)
    for i in range(4):
        type(Key.DOWN)
        wait(1)
    type(Key.ENTER)
    print("execute shutdown")
    wait(20)
    if exists():
        print("open login window")
    else:
        print("open login window FAIL")
    type(target["ac"])
    wait(1)
    type(Key.TAB)
    wait(1)
    type(target["pwd"])
    wait(1)
    type(Key.ENTER)
    wait(5)
    print("input ac & pwd")
    wait(3)
    if exists():
        print("start to shutdown")
        flag = "True"
        msg_region = Region(Region(481,270,320,138))
        msg_region.click()
        print("close shutdown dia")
    else:
        print("Fail to start shutdown")
        flag = "False"
    with open("result.txt", "w") as fp:
        fp.write(flag)
    print("--- End " + fun_name + " ---")

if __name__ == "__main__":
    qfinder_shutdown()
