def qfinder_network_driver():
    fun_name = sys._getframe().f_code.co_name
    print("*** Start to " + fun_name + " ***")
    # open qfinder
    open_qfinder()
    click()
    print("click refresh button")
    wait_please(loop=10, time=10)
    #  find target NAS
    find_target_nas(name = target["name"], lanip1 = target["lanip1"])
    wait(1)

    click()
    print("click network drivers button")
    wait(10)
    if exists():
        print("find input network drive window")
        type(target["ac"])
        wait(1)
        type(Key.TAB)
        wait(1)
        type(target["pwd"])
        wait(1)
        type(Key.ENTER)
        wait(5)
        print("input ac & pwd")
    else:
        pass

    if exists(Pattern().similar(0.80)):
        click(Pattern().similar(0.80))
        print("max window")
        wait(2)
        a = Region(Region(94,0,92,19))
        t = a.text()
        print(t)
        if t == target["lanip1"]:
            print("open network drivers PASS")
            flag = "True"
        else:
            print("open network drivers FAIL")
            flag = "False"
        click(Pattern().similar(0.80))

    elif exists():
        print("open network drivers FAIL")
        flag = "False"
    else:
        flag = "False"

    with open("result.txt", "w") as fp:
       fp.write(flag)
    print("--- End " + fun_name + " ---")

if __name__ == "__main__":
    qfinder_network_driver()
